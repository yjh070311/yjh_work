class Solution(object):

    def findAnagrams(self, s, p):

        """

        :type s: str

        :type p: str

        :rtype: List[int]

        """

        current_hash={}

        request_hash={}

        result=[]

        n=len(p)

        for char in p:

            request_hash[char]+=1

        for right in s:

            left=right-n+1

            current_hash[right]+=1

            if current_hash==request_hash:

                result.append(left)

            if left>0:

                current_hash[left]-=1

        return result
