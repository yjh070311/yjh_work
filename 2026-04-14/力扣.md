class Solution(object):

    def findAnagrams(self, s, p):

        """

        :type s: str

        :type p: str

        :rtype: List[int]

        """

        current_hash={}

        request_hash={}

        result=[]

        n=len(p)

        for char in p:

            request_hash[char]+=1

        for right in s:

            left=right-n+1

            current_hash[right]+=1

            if current_hash==request_hash:

                result.append(left)

            if left>0:

                current_hash[left]-=1

        return result

def findAnagrams(s: str, p: str) -> List[int]:
    """
    优化版：使用差值计数，避免每次都比较整个数组
    """
    if len(s) < len(p):
        return []
    
    result = []
    p_len = len(p)
    s_len = len(s)
    
    # 使用差值数组：diff[char] = count_in_p - count_in_window
    diff = [0] * 26
    
    # 初始化：统计 p 的字符（正数）
    for char in p:
        diff[ord(char) - ord('a')] += 1
    
    # 初始化：减去窗口中的字符（负数）
    for i in range(p_len):
        diff[ord(s[i]) - ord('a')] -= 1
    
    # 检查第一个窗口
    if all(x == 0 for x in diff):
        result.append(0)
    
    # 滑动窗口
    for i in range(p_len, s_len):
        # 移除左边字符
        left_char = s[i - p_len]
        diff[ord(left_char) - ord('a')] += 1
        
        # 加入右边字符
        right_char = s[i]
        diff[ord(right_char) - ord('a')] -= 1
        
        # 检查是否全零
        if all(x == 0 for x in diff):
            result.append(i - p_len + 1)
    
    return result
